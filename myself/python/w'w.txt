# https://view.inews.qq.com/g2/getOnsInfo?name=disease_h5
import json
import requests

jsonStr = requests.get('https://view.inews.qq.com/g2/getOnsInfo?name=disease_h5').text
jsonStr = json.loads(jsonStr)['data']
obj = json.loads(jsonStr)
print('统计时间：',obj['lastUpdateTime'])
print('确诊：',obj['chinaTotal']['confirm'])
print('疑似：',obj['chinaTotal']['suspect'])
print('治愈：',obj['chinaTotal']['heal'])
print('死亡：',obj['chinaTotal']['dead'])
